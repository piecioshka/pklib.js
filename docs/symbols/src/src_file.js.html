1 /**
  2  * @package pklib.file, pklib.string
  3  */
  4 
  5 /**
  6  * JS file loader
  7  */
  8 pklib.file = (function () {
  9     "use strict";
 10 
 11     var copy_files = [];
 12 
 13     /**
 14      * @param {String} url
 15      * @param {Function} callback
 16      */
 17     function simple_load_js(url, callback) {
 18         /**
 19          * Create HTMLElement <script>
 20          */
 21         var script = document.createElement("script");
 22         script.type = "text/javascript";
 23         script.src = url;
 24 
 25         if (script.readyState === undefined) {
 26             /**
 27              * Method run when request has ended
 28              */
 29             script.onload = function () {
 30                 if (typeof callback === "function") {
 31                     callback(script);
 32                 }
 33             };
 34         } else {
 35             /**
 36              * Method run when request has change state
 37              */
 38             script.onreadystatechange = function () {
 39                 if (script.readyState === "loaded" || script.readyState === "complete") {
 40                     script.onreadystatechange = null;
 41                     if (typeof callback === "function") {
 42                         callback(script);
 43                     }
 44                 }
 45             };
 46         }
 47 
 48         if (document.head === undefined) {
 49             document.head = document.getElementsByTagName("head")[0];
 50         }
 51 
 52         document.head.appendChild(script);
 53     }
 54 
 55     /**
 56      * Load JS files. Url to files could be with path absolute or not.
 57      * If you must load more than 1 file use array, to set url to files
 58      *
 59      * @param {String|Array} files
 60      * @param {Function} callback
 61      */
 62     function load_js_file(files, callback) {
 63         var file;
 64 
 65         if (typeof files === "string") {
 66             file = files;
 67             simple_load_js(file, function (script) {
 68                 if (typeof callback === "function") {
 69                     callback(script);
 70                 }
 71             });
 72         } else if (pklib.array.is_array(files)) {
 73             if (!copy_files.length) {
 74                 copy_files = pklib.object.mixin(copy_files, files);
 75             }
 76 
 77             file = files.shift();
 78 
 79             if (file === undefined) {
 80                 if (typeof callback === "function") {
 81                     callback({
 82                         src: copy_files[copy_files.length - 1]
 83                     });
 84 
 85                     copy_files = [];
 86                 }
 87             } else {
 88                 simple_load_js(file, function () {
 89                     load_js_file(files, callback);
 90                 });
 91             }
 92         } else {
 93             throw new Error("pklib.file.loadjs: @files not {String} or {Array}");
 94         }
 95     }
 96 
 97     // exports
 98     return {
 99         loadjs: load_js_file
100     };
101 }());
