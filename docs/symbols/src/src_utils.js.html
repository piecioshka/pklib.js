 13     /**
 14      * @private
 15      * @function
 16      * @param {Event} evt
 17      */
 18     function open_trigger(evt) {
 19         var url = "";
 20 
 21         if (evt.originalTarget &&
 22                 typeof evt.originalTarget === "object" &&
 23                 evt.originalTarget.href !== undefined) {
 24             url = evt.originalTarget.href;
 25         } else if (evt.toElement &&
 26                 typeof evt.toElement === "object" &&
 27                 evt.toElement.href !== undefined) {
 28             url = evt.toElement.href;
 29         } else if (evt.srcElement &&
 30                 typeof evt.srcElement === "object" &&
 31                 evt.srcElement !== undefined) {
 32             url = evt.srcElement.href;
 33         }
 34 
 35         window.open(url);
 36 
 37         try {
 38             evt.preventDefault();
 39         } catch (ignore) {
 40             window.event.returnValue = false;
 41         }
 42 
 43         return false;
 44     }
 45 
 46     /**
 47      * @private
 48      * @function
 49      * @param {HTMLElement} obj
 50      */
 51     function clear_focus(obj) {
 52         if (pklib.dom.is_element(obj)) {
 53             pklib.event.add(obj, "focus", function () {
 54                 if (obj.value === obj.defaultValue) {
 55                     obj.value = "";
 56                 }
 57             });
 58             pklib.event.add(obj, "blur", function () {
 59                 if (obj.value === "") {
 60                     obj.value = obj.defaultValue;
 61                 }
 62             });
 63         }
 64     }
 65 
 66     /**
 67      * @private
 68      * @function
 69      * @param {HTMLElement} [area]
 70      */
 71     function outerlink(area) {
 72         var i, len,
 73             link, links;
 74 
 75         area = area || document;
 76 
 77         links = pklib.dom.by_tag("a", area);
 78         len = links.length;
 79 
 80         for (i = 0; i < len; ++i) {
 81             link = links[i];
 82             if (link.rel === "outerlink") {
 83                 pklib.event.add(link, "click", open_trigger.bind(link));
 84             }
 85         }
 86     }
 87 
 88     /**
 89      * @private
 90      * @function
 91      * @param {HTMLElement} element
 92      * @param {String} [text]
 93      */
 94     function confirm(element, text) {
 95         var response;
 96         if (element !== undefined) {
 97             text = text || "Sure?";
 98 
 99             pklib.event.add(element, "click", function (evt) {
100                 response = window.confirm(text);
101                 if (!response) {
102                     try {
103                         evt.preventDefault();
104                     } catch (ignore) {
105                         window.event.returnValue = false;
106                     }
107 
108                     return false;
109                 }
110                 return true;
111             });
112         }
113     }
