1 /**
  2  * @package pklib.css
  3  * @dependence pklib.string. pklib.dom
  4  */
  5 (function (global) {
  6     "use strict";
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9         /**
 10          * RegExp use to delete white chars
 11          * @type RegExp
 12          */
 13         rclass = /[\n\t\r]/g,
 14         /**
 15          * Check typeof params
 16          * @private
 17          * @function
 18          * @param {String} cssClass
 19          * @param {HTMLElement} element
 20          * @throws {TypeError} If first param is not string, or second param is not Node
 21          */
 22         checkParams = function (cssClass, element, callFuncName) {
 23             var prefix = "pklib.css." + callFuncName;
 24             if (typeof cssClass !== "string") {
 25                 throw new TypeError(prefix + ": @cssClass: not {String}");
 26             }
 27             if (!pklib.dom.isNode(element)) {
 28                 throw new TypeError(prefix + ": @element: not {HTMLElement}");
 29             }
 30         },
 31         /**
 32          * Utils method related css on tags in DOM tree
 33          * @namespace
 34          */
 35         css = {
 36             /**
 37              * Add CSS class to element define in second parameter
 38              * @memberOf css
 39              * @function
 40              * @param {String} cssClass
 41              * @param {HTMLElement} element
 42              * @throws {TypeError} If first param is not string, or second param is not Node
 43              */
 44             addClass: function (cssClass, element) {
 45                 checkParams(cssClass, element, "addClass");
 46                 var classElement = element.className;
 47                 if (!pklib.css.hasClass(cssClass, element)) {
 48                     if (classElement.length) {
 49                         classElement += " " + cssClass;
 50                     } else {
 51                         classElement = cssClass;
 52                     }
 53                 }
 54                 element.className = classElement;
 55             },
 56             /**
 57              * Remove CSS class from element define in second parameter
 58              * @memberOf css
 59              * @function
 60              * @param {String} cssClass
 61              * @param {HTMLElement} element
 62              * @throws {TypeError} If first param is not string, or second param is not Node
 63              */
 64             removeClass: function (cssClass, element) {
 65                 checkParams(cssClass, element, "removeClass");
 66                 var regexp = new RegExp("(\\s" + cssClass + ")|(" + cssClass + "\\s)|" + cssClass, "i");
 67                 element.className = pklib.string.trim(element.className.replace(regexp, ""));
 68             },
 69             /**
 70              * Check if element has CSS class
 71              * @memberOf css
 72              * @function
 73              * @param {String} cssClass
 74              * @param {HTMLElement} element
 75              * @throws {TypeError} If first param is not string, or second param is not Node
 76              * @returns {Boolean}
 77              */
 78             hasClass: function (cssClass, element) {
 79                 checkParams(cssClass, element, "hasClass");
 80                 var className = " " + cssClass + " ";
 81                 return ((" " + element.className + " ").replace(rclass, " ").indexOf(className) > -1);
 82             }
 83         };
 84 
 85     pklib.css = css;
 86 }(this));
