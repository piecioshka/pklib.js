1 /**
  2  * @package pklib.ui.size
  3  * @dependence pklib.string
  4  */
  5 
  6 /**
  7  * Check ui dimensions
  8  */
  9 pklib.ui.size = (function () {
 10     "use strict";
 11 
 12     /**
 13      * @param {String} name
 14      * @throws {TypeError}
 15      * @returns {Number}
 16      */
 17     function size_of_window(name) {
 18         var clientName;
 19         pklib.common.assert(typeof name === "string", "pklib.ui.size.window: @name: not {String}");
 20 
 21         name = pklib.string.capitalize(name);
 22         clientName = document.documentElement["client" + name];
 23         return (document.compatMode === "CSS1Compat" && clientName) ||
 24             document.body["client" + name] ||
 25             clientName;
 26     }
 27 
 28     /**
 29      * @param {String} name
 30      * @returns {Number}
 31      */
 32     function size_of_document(name) {
 33         var clientName,
 34             scrollBodyName,
 35             scrollName,
 36             offsetBodyName,
 37             offsetName;
 38 
 39         pklib.common.assert(typeof name === "string", "pklib.ui.size.document: @name: not {String}");
 40 
 41         name = pklib.string.capitalize(name);
 42         clientName = document.documentElement["client" + name];
 43         scrollBodyName = document.body["scroll" + name];
 44         scrollName = document.documentElement["scroll" + name];
 45         offsetBodyName = document.body["offset" + name];
 46         offsetName = document.documentElement["offset" + name];
 47         return Math.max(clientName, scrollBodyName, scrollName, offsetBodyName, offsetName);
 48     }
 49 
 50     /**
 51      * @param {HTMLElement} obj
 52      * @param {String} name
 53      * @returns {Number}
 54      */
 55     function size_of_object(obj, name) {
 56         pklib.common.assert(typeof name === "string", "pklib.ui.size.object: @name: not {String}");
 57         pklib.common.assert(pklib.dom.is_element(obj), "pklib.ui.size.object: @obj: not {HTMLElement}");
 58 
 59         name = pklib.string.capitalize(name);
 60         var client = obj["client" + name],
 61             scroll = obj["scroll" + name],
 62             offset = obj["offset" + name];
 63         return Math.max(client, scroll, offset);
 64     }
 65 
 66     // public APi
 67     return {
 68         window: size_of_window,
 69         document: size_of_document,
 70         object: size_of_object
 71     };
 72 }());
