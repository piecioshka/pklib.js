1 /**
  2  * @package pklib.ui.loader
  3  * @dependence pklib.dom, pklib.event, pklib.utils
  4  */
  5 (function (global) {
  6     "use strict";
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9         document = global.document || {},
 10         id = "pklib-loader-wrapper",
 11         settings = {
 12             src: "",
 13             container: null,
 14             style: {
 15                 width: 31,
 16                 height: 31,
 17                 zIndex: 1010
 18             },
 19             center: true
 20         };
 21 
 22     /**
 23      * Loader adapter.
 24      * Show animate image (GIF) on special place.
 25      * @namespace
 26      */
 27     pklib.ui.loader = {
 28         obj_id: id,
 29         /**
 30          * @memberOf pklib.ui.loader
 31          * @function
 32          * @param {object} config
 33          * @param {function} callback
 34          */
 35         show: function (config, callback) {
 36             settings.container = document.body;
 37             settings = pklib.object.mixin(settings, config);
 38 
 39             var loader = document.createElement("img"),
 40                 loaderStyle = loader.style,
 41                 style;
 42 
 43             loader.setAttribute("id", this.obj_id);
 44             loader.setAttribute("src", settings.src);
 45 
 46             for (style in settings.style) {
 47                 if (settings.style.hasOwnProperty(style)) {
 48                     loaderStyle[style] = settings.style[style];
 49                 }
 50             }
 51 
 52             if (settings.center) {
 53                 pklib.ui.center(loader, settings.container);
 54 
 55                 pklib.event.add(global, "resize", function () {
 56                     pklib.ui.center(loader, settings.container);
 57                 });
 58             }
 59 
 60             settings.container.appendChild(loader);
 61 
 62             if (typeof callback === "function") {
 63                 callback();
 64             }
 65             // clear memory
 66             loader = null;
 67         },
 68         /**
 69          * @memberOf pklib.ui.loader
 70          * @function
 71          * @param {Function} callback
 72          * @returns {Boolean}
 73          */
 74         close: function (callback) {
 75             var loader = pklib.dom.by_id(this.obj_id),
 76                 result = false;
 77             if (loader !== null) {
 78                 loader.parentNode.removeChild(loader);
 79                 this.close(callback);
 80                 result = true;
 81             }
 82             if (typeof callback === "function") {
 83                 callback();
 84             }
 85             return result;
 86         }
 87     };
 88 }(this));
